CREATE TABLE peliculas(

idPel int identity(1,1) primary key,

titulo VARCHAR(255) NOT NULL,

year SMALLINT NOT NULL

)

CREATE TABLE cines(

idCine int identity(1,1) primary key,

nombre VARCHAR(255) NOT NULL,

direc VARCHAR(255) NOT NULL,

)

CREATE TABLE pelicula_cines(

idPc int identity(1,1) primary key,

idPel int references peliculas,

idCine int references cines,

)
